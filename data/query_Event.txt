PREFIX ns: <http://www.semanticweb.org/rayenbourguiba/ontologies/2023/9/untitled-ontology-4#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>

SELECT DISTINCT ?event ?title ?description ?date ?address ?capacity ?eventType
WHERE {
  ?event rdf:type ns:OnlineEvent;
         ns:title ?title;
         ns:description ?description;
         ns:date ?date;
         ns:address ?address;
         ns:capacity ?capacity.
  BIND("Online Event" AS ?eventType)
}
